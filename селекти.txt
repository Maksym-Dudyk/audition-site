#  Всі відгуки від одної людини за певний період з 22 квітня по 25 квітня 2023 року
 SELECT  feedback.name, feedback.content, feedback.date, user.first_name, user.last_name, tariff.name
   FROM feedback
     JOIN user
     ON user.id = feedback.user_id
   JOIN tariff
     ON tariff.id = feedback.tariff_id
   WHERE user.first_name = "Настя"
     AND user.last_name = "Крук"
     AND feedback.date BETWEEN "2012-04-22" AND "2022-04-25";
    
# Всі відгуки за 2022 рік
 SELECT  feedback.name, feedback.content, feedback.date, user.id, user.first_name, user.last_name, tariff.name
   FROM feedback
     JOIN user
     ON user.id = feedback.user_id
   JOIN tariff
     ON tariff.id = feedback.tariff_id
   WHERE feedback.date REGEXP '2022';

# Всі відгуки клієнта з іменем Олександр Пристай
 SELECT  feedback.name, feedback.content, feedback.date, user.id, user.first_name, user.last_name, tariff.name
    FROM feedback
      JOIN user
      ON user.id = feedback.user_id
    JOIN tariff
      ON tariff.id = feedback.tariff_id
    WHERE user.first_name = "Олександр"
      AND user.last_name = "Пристай";

#  Перегляд відгуку з наступними даними: Ім'я та прізвище клієнта, дата відгуку, тариф клаєнта та безпосередньо сам відгук

# Перегляд інформаційних публікацій з фотографіями та типами публікацій
 SELECT post.text, post.img_link, post_type.name AS "Post type"
   FROM post
     JOIN post_type
     ON post_type.id = post.type_id;
